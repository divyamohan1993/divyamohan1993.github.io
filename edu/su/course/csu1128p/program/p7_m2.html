#include <stdio.h>
int main()
{
    int ra, ca, rb, cb, c = 0, num, num2, num3, e;
    printf("enter no of rows of matrix 1:");
    scanf("%d", &ra);

    printf("enter no of column of matrix 1:");
    scanf("%d", &ca);

    printf("enter no of rows of matrix 2:");
    scanf("%d", &rb);

    printf("enter no of column of matrix 2:");
    scanf("%d", &cb);

    int a[ra][ca], b[rb][cb];
    if (ra == rb && ca == cb)
    {
        for (int i = 0; i < ra; i++)
        {
            for (int j = 0; j < ca; j++)
            {
                printf("a%d%d:", i, j);
                scanf("%d", &a[i][j]);
            }
        }
        for (int i = 0; i < rb; i++)
        {
            for (int j = 0; j < cb; j++)
            {
                printf("b%d%d:", i, j);
                scanf("%d", &b[i][j]);
            }
        }
        e = ra * cb;
        if (a[0][0] > b[0][0] || a[0][0] == b[0][0])
        {
            num2 = a[0][0] / b[0][0];
            for (int i = 0; i < ra; i++)
            {
                for (int j = 0; j < ca; j++)
                {
                    num = a[i][j] / b[i][j];
                    if (num * b[i][j] == a[i][j])
                    {
                        num3 = num3 + num;
                        c++;
                    }
                }
            }
            if (c == e && num2 == num3 / e)
            {
                printf("identical matrix\n");
            }
            else
            {
                printf("non-identical matrix\n");
            }
        }
        if (b[0][0] > a[0][0])
        {
            num2 = b[0][0] / a[0][0];
            for (int i = 0; i < ra; i++)
            {
                for (int j = 0; j < ca; j++)
                {
                    num = b[i][j] / a[i][j];
                    if (num * a[i][j] == b[i][j])
                    {
                        num3 = num3 + num;
                        c++;
                    }
                }
            }
            if (c == e && num2 == num3 / e)
            {
                printf("identical matrix\n");
            }
            else
            {
                printf("non-identical matrix\n");
            }
        }
    }
    else
    {
        printf("non identical matrix\n");
    }
    printf(" %d %d %d %d %d", num, num2, num3, e, c);
    return 0;
}
