#include <stdio.h>
#include <math.h>
{
    int i;
    printf("This program prints all four digit perfect squares.\n");
    for (i = 999; i <= 9999; i++)
    {
        int s = sqrt(i);
        if (s * s == i)
        {
            printf("\t %d", i);
        }
    }
    printf("\n\n");
    return 0;
}
