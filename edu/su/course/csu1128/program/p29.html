#include <stdio.h>
int main()
{
    int d, m, j;

    printf("\n\n Program to demonstrate the nested for loop by printing a right angled triangle. \n\n");

    printf("Enter total number of times you want the loop to repeat - (Example: 5) - and press enter: ");
    scanf("%d", &d);

    for (m = 1; m <= d; m++)
    {
        for (j = 1; j <= m; j++)
        {
            printf("%d ", j);
        }
        printf("\n");
    }
    printf("\n\n");
    return 0;
}
