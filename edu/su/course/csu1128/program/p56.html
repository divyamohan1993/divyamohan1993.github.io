#include <stdio.h>
int main()
{
    printf("\n\n Program for swapping of two numbers entered by the user through call by value method. \n\n");

    int a, b, temp;

    printf("Enter the value for a and b: ");
    scanf("%d %d", &a, &b);

    printf("Before: a = %d, b = %d\n", a, b);

    temp = a;
    a = b;
    b = temp;

    printf("After: a = %d, b = %d\n", a, b);

    return 0;
}

Output:

Enter the value for a and b: 12 43
Before: a = 12, b = 43
After: a = 43, b = 12
