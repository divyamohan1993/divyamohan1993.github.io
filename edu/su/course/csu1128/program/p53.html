#include <stdio.h>

int main()
{
    printf("\n\n Program to check the use of pointers an its use of address space. \n\n");
    
    int d;
    printf("Enter the value: ");
    scanf("%d", &d);
    
    printf("\n\n Number (d) is defined as %d. \n Its address (&d) is %p.\n Points to value at that address (*(&d)) is %i
    \n\n", d, &d, *(&d));
    return 0;
}

Output:
