#include <stdio.h>
int main()
{
    printf("\n\n Program to take the array size and its elments from the user and sort them. Example:3, 2, 9, 1 = 1, 2, 3, 9\n\n");

    // Read the size of the array from the user
    printf("Enter the size of the array: ");
    int d;
    scanf("%d", &d);

    // Declare an array of integers with d elements
    int e[d];

    // Read the elements of the array from the user
    printf("Enter the elements of the array:\n");
    for (int i = 0; i < d; i++)
    {
        scanf("%d", &e[i]);
    }

    // Print the original array
    printf("Original array: ");
    for (int i = 0; i < d; i++)
    {
        printf("%d ", e[i]);
    }
    printf("\n");

    // Sort the array in ascending order using a for loop
    for (int m = 0; m < d - 1; m++)
    {
        // Find the minimum element in the unsorted part of the array
        int min_index = m;
        for (int j = m + 1; j < d; j++)
        {
            if (e[j] < e[min_index])
            {
                min_index = j;
            }
        }

        // Swap the minimum element with the first element of the unsorted part
        int temp = e[m];
        e[m] = e[min_index];
        e[min_index] = temp;
    }

    // Print the sorted array
    printf("Sorted array: ");
    for (int i = 0; i < d; i++)
    {
        printf("%d ", e[i]);
    }
    printf("\n");

    return 0;
}
