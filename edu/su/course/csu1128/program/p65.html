#include <stdio.h>
#define f(x) x * x * x
int main()
{
    int x;

    printf("\n\n Program to demonstrate the macro by finding the cube of the value entered by the user. \n\n");

    printf("Enter a value for x: ");
    scanf("%d", &x);

    // Use the macro f(x) to calculate x*x
    int result = f(x);

    printf("f(%d) = %d\n", x, result);

    return 0;
}
