/*  Miscellaneous Program by dmj.one
 *  DO NOT COPY     DO NOT REPRODUCE    DO NOT MODIFY
 *  
 * (c) 2023, Divya Mohan for dmj.one. All rights reserved.
 */

#include <stdio.h>
void func(int a[], int n) {
    int i;
    for (i = 0; i < n; i++) {
        a[i] = a[i] + 2;
        a++;
    }
}

int main() {
    int a[] = {1, 2, 3, 4};
    int n = sizeof(a) / sizeof(int);
    func(a, n);
    int i;
    for (i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }
    return 0;
}

Output:

  3 5 7 9 / 3 2 5 4? You decide!
